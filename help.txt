# build docker image
docker build -t build_messenger .

# delete image
docker image rm build_messenger

# run container
docker run --rm -it --name server (or client) --security-opt seccomp=unconfined build_messenger

# run and rebuild docker-compose
docker compose up -d --build

# shut down docker-compose containers
docker compose down

# compile with clang-tidy check
rm -rf *; cmake -DCLANG_TIDY_CHECK=1 -DCMAKE_TOOLCHAIN_FILE=../CMakeToolchain.cmake ..; cmake --build . -j 4

# run clang-format
./tools/style-formatter.sh
